La lógica para dar la respuesta al usuario deberá estar incluida en una función a la que se llamará tras cada intento.

5
bajo

bajo

acertado

# De random importa randint
from random import randint

# Defina la función
def adivinansa(aleatorio):
    # Construya el cuerpo del código
    num = int(input())
    if num == aleatorio:
        print("acertado")
    elif num < aleatorio:
        print("bajo")
        adivinansa(aleatorio)
    else:
        print("alto")
        adivinansa(aleatorio)
